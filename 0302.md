### 两个栈实现队列
```
public class CQueue {
    int[] stack1;
    int[] stack2;
    int length1;
    int length2;
    //用栈的先进后出实现队列的先进先出
    //可以存入一个栈，要取出时，再放入另一个栈中，再出栈
    public CQueue() {
        stack1=new int[11000];
        stack2=new int[11000];
        length1=0;
        length2=0;
    }

    public void appendTail(int value) {
        stack1[length1]=value;
        length1++;
    }

    public int deleteHead() {
        if(length2!=0){
            return stack2[--length2];
        }
        if(length1==0) return -1;
        for(int i=length1-1,j=0;i>=0;i--,j++){
            stack2[j]=stack1[i];
            length2=j;
        }
        length1=0;
        return stack2[length2];
    }
}
```

### 斐波那契数列
```
	public int fib(int n) {
        if(n==0) return 0;
        if(n==1) return 1;
        //递归超时，改用动态规划，时间复杂度由O(n^2)变为O(n),空间复杂度为O(n)
        int[] fibs=new int[n+1];
        fibs[1]=1;
        for(int i=2;i<=n;i++){
            fibs[i]=(fibs[i-1]+fibs[i-2])%1000000007;
        }
        return fibs[n];
    }
```

### 青蛙跳台阶问题
```
	public int numWays(int n) {
        if(n==0) return 1;
        if(n<=2) return n;
        //动态规划
        int[] dp=new int[n+1];
        dp[1]=1;
        dp[2]=2;
        //在第n阶可由n-1阶跳一步，和在第n-2阶跳两步的可能跳法组成
        for(int i=3;i<=n;i++){
            dp[i]=(dp[i-1]+dp[i-2])%1000000007;
        }
        return dp[n];
    }
```
